<script>
        document.addEventListener("DOMContentLoaded", () => {
            // Define the URL to redirect after consent
      const url = '/job-seeker/profile/basic';
      // Configuration object mapping button IDs to their respective redirect URLs
        const config = {

            showAIModalForEditProfile: {
                url:url
            },
            showAIModalForReview: {
                url: '/job-seeker/profile/review'
            },
            showAIModalForBuildProfile: {
                url: url
            },
        };
         // DOM element references
        const modal = document.getElementById('aiModal');
        const okBtn = document.getElementById('aiModalOkBtn');
        //const titleEl = document.getElementById('aiTitle');
        const bodyEl = document.getElementById('aiBody');
         // API endpoints for consent check and consent logging
        const logConsentUrl = '/JobSeekerProfile/LogAIConsent';
        const checkConsentUrl = '/JobSeekerProfile/CheckAIConsent';
        const modalContent = modal.querySelector(".ai-modal-content");

        Object.entries(config).forEach(([btnId, { url }]) => {
            const btn = document.getElementById(btnId);
            if (!btn) return;

            btn.addEventListener('click', () => {
                // First, check if consent already exists
                fetch(checkConsentUrl, {
                    method: 'GET',
                    headers: {
                        'Content-Type': 'application/json'
                    },

                })
                    .then(res => res.json())
                    .then(data => {
                        if (data.consent) {
                            // Consent exists, redirect directly
                            window.location.href = url;
                        } else {
                           // No consent found - load and show AI Transparency modal
                            fetch('/ai-transparency.json')
                                .then(res => res.json())
                                .then(data => {
                                    debugger;
                                    // Remove any old heading if it exists
                                    const existingTitle = modalContent.querySelector('h2');
                                    if (existingTitle) {
                                        existingTitle.remove();
                                    }

                                    // Only create <h2> if title exists
                                    if (data.title && data.title.trim() !== '') {
                                        const h2 = document.createElement('h2');
                                        h2.id = "aiTitle"
                                        h2.textContent = data.title;
                                        modalContent.prepend(h2);
                                    }

                                    bodyEl.innerHTML = data.content;
                                    okBtn.setAttribute('data-redirect-url', url); // Store redirect URL
                                    modal.style.display = 'flex';
                                })
                                .catch(err => console.error("Modal fetch error:", err));
                        }
                    })
                    .catch(err => console.error("Consent check error:", err));
            });
        });
         // Handle "OK" button click inside modal
        okBtn.addEventListener('click', () => {
            //hide the modal
            modal.style.display = 'none';

           // Step 2: Log user's AI consent
            fetch(logConsentUrl, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    timestamp: new Date().toISOString(),
                })
            })
                .then(() => {
                     // Redirect after consent is successfully logged
                    const redirectUrl = okBtn.getAttribute('data-redirect-url');
                    if (redirectUrl) {
                        window.location.href = redirectUrl;
                    }
                })
                .catch(err => console.error("Consent save error:", err));
        });
    });
</script>

        function getCookie(cname) {
            var name = cname + "="; var decodedCookie = decodeURIComponent(document.cookie); var ca = decodedCookie.split(';'); for (var i = 0; i < ca.length; i++) {
                var c = ca[i]; while (c.charAt(0) == ' ') { c = c.substring(1) }
                if (c.indexOf(name) == 0) { return c.substring(name.length, c.length) }
            }
            return ""
